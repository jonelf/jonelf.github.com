fisherYates = (arr) ->
  swapFrom = arr.length
  if swapFrom == 0 then return false

  while --swapFrom
    swapTo = Math.floor(Math.random() * (swapFrom+1))
    [arr[swapFrom], arr[swapTo]] = [arr[swapTo], arr[swapFrom]]
